def generate_data_and_pvals(rho=0):
    mus = np.append(np.zeros(m0), mu * np.ones(m - m0))
    s = np.random.normal(0, 1, m)
    s = rho * s[0] + (1 - rho) * s + mus
    return s, norm.sf(s)

def bh_histograms(n=5000, alpha=0.1):
    r, v, q = np.zeros(n), np.zeros(n), np.zeros(n)
    for rho in [0, 0.95]:
        for i in range(n):
            _, p_vals = generate_data_and_pvals(rho=rho)
            rejected, _, _, _ = statsmodels.stats.multipletests(p_vals, alpha=alpha, method='fdr_bh')
            r[i] = max(np.count_nonzero(rejected), 1)
            v[i] = np.count_nonzero(rejected[:m0])

        plt.hist(r, bins=50)
        plt.title(f"Histogram of R with rho={rho}")
        plt.show()
        
        plt.hist(v)
        plt.title(f"Histogram of V with rho={rho}")
        plt.show()
        
        q = v / r
        plt.hist(q, bins=50)
        plt.title(f"Histogram of Q with rho={rho}")
        plt.show()

Regarding R, we can see that for rho=0 the number of rejections distributes like a Gaussian with a center in 280 approximately. When rho=0.95 the number of rejections is distributed in a very different way, while there are only 3 options: reject almost all the hypotheses (R=1000), reject half of them (R=500), or reject almost none of them (R=0).

Regarding V, when rho=0 we have the number of false rejections that distributes like a Gaussian, with a center in 15 approximately. For rho=0.95 V distributes with two options: no false rejections (V=0) or rejection of all the null hypotheses (V=500).

Regarding Q, when rho=0 the relation V/R+ also looks like a Gaussian, and its center is in 0.05. When rho=0.95 we have two main values for Q: Q=0, which means there was no False positive, or Q=0.5, which means a very high number of false positive.

In all the graphs we can see that when rho=0.95 we get a distribution with higher variance.

def bh_graph(n=5000, alpha=0.1):
    rhos = np.arange(0, 1.05, 0.05)
    means = np.zeros(len(rhos))
    stds = np.zeros(len(rhos))

    for j, rho in enumerate(rhos):
        q = np.zeros(n)
        for i in range(n):
            _, p_vals = generate_data_and_pvals(rho=rho)
            rejected, _, _, _ = statsmodels.stats.multipletests(p_vals, alpha=alpha, method='fdr_bh')
            q[i] = np.count_nonzero(rejected[:m0]) / max(np.count_nonzero(rejected), 1)
        means[j] = np.mean(q)
        stds[j] = np.std(q)

    plt.errorbar(rhos, means, stds, linestyle='None', marker='^')
    plt.xticks(rhos)
    plt.title("mean and standard deviation as a function of rho")
    plt.savefig("mean_std_graph")
    plt.show()

def histogram_and_densities(rho=0, pi_0=0.5):
    x, _ = generate_data_and_pvals(rho=rho)
    plt.hist(x, bins=100, density=True, label="data")
    x_axis = np.linspace(-4, 4, 1000)
    f0 = norm.pdf(x_axis, 0, 1)
    f1 = norm.pdf(x_axis, 2, 1)
    plt.plot(x_axis, pi_0 * f0, label="F0")
    plt.plot(x_axis, (1-pi_0) * f1, label="F1")
    plt.plot(x_axis, pi_0 * f0 + (1-pi_0) * f1, label=f"{pi_0}*F0 + {(1-pi_0)}*F1")
    plt.legend()
    plt.show()

def fdr_empirical(rho=0):
    epsilon = 0.5
    x, p_vals = generate_data_and_pvals(rho=rho)
    z = np.linspace(-4, 4, m)[:, np.newaxis]

    FDRs_mat = np.broadcast_to(x, (z.size, x.size)) > z
    fdrs_mat = np.logical_and(np.broadcast_to(x, (z.size, x.size)) > z - 
                              epsilon, np.broadcast_to(x, (z.size, x.size)) < z + epsilon)

    for i, mat in enumerate([FDRs_mat, fdrs_mat]):
        r = np.maximum(np.sum(mat, axis=1), 1)
        v = np.sum(mat[:, :m0], axis=1)
        q = v / r
        z = z[np.where(~np.isnan(q))]
        q = q[~np.isnan(q)]
        plt.plot(z, q, 'o', label='FDR' if i == 0 else 'fdr')

    plt.title("FDR vs. fdr empirical")
    plt.legend()
    plt.show()

def q_values(rho=0, pi_0=0.5):
    x, p_vals = generate_data_and_pvals(rho=rho)
    sorted_p_vals = np.sort(p_vals)
    q_vals = np.minimum((sorted_p_vals * m) / np.arange(1, m+1), 1)
    for i in range(m-2, -1, -1):
        q_vals[i] = min(q_vals[i], q_vals[i+1])
    z = -norm.ppf(sorted_p_vals)
